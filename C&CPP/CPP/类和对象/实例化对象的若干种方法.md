```CPP
#include <iostream>
using String = std::string;
using std::cout;
class People
{
public:
	int m_num;
public:
	People()
		: m_num(0) {}
	People(int num)
		: m_num(num) {}
	void Print() const
	{
		cout << m_num << "\n";
	}
};
int main()
{
    cout << "--- 开始创建对象 ---\n";

    People p1;                 // 默认构造
    People p2(2);              // 参数构造
    People p3 = People(3);     // 参数构造
    People p4 = p2;            // 拷贝构造
    People p5 = 4;             // 隐式转换：参数构造
    People* p6 = new People(9); // 参数构造（堆上）
    People p7{ 90 };             // 统一初始化：参数构造
    People* p8 = new People[10]{}; // 10次默认构造（堆上数组）
	//People p9();    // ⚠️这是函数声明：声明一个名为p9、返回People类型、不接受参数的函数
	People p10{};   // 使用统一初始化（C++11），创建People对象并进行值初始化

    cout << "\n--- 输出值 ---\n";
    p1.Print();
    p2.Print();
    p3.Print();
    p4.Print();
    p5.Print();
    p6->Print();
    p7.Print();
    // 数组中的元素未初始化，需要循环打印...

    cout << "\n--- 清理内存 ---\n";
    delete p6;                  // 析构 9
    delete[] p8;                // 析构数组中的10个对象

    cout << "--- 程序结束，栈对象自动析构 ---\n";
    // p1-p7 在这里自动析构
}
```
## 栈中创建对象
除了`p6`和`p8`，其他的都是栈上的对象。他们的生命周期由程序自动控制。但是如果较大或者数量较多，容易发生栈溢出，此时可以考虑在堆上生成。
## 堆中创建对象
`p6`和`p8`是在堆上的对象。内存中的堆区较大，适合存放较多或较大的对象。他们的生命周期很长，需要手动释放，否则容易发生内存溢出。请注意，`new`返回的是一个指针，需要存在指针变量中。
