这是一个简单的单参数的类模板示例：
```CPP
#include <iostream>

template<typename T>
class Box
{
private:
	T m_val;
public:
	Box(const T& val)
		: m_val(val) {}

	T GetVal() const { return m_val; }
	void SetVal(const T& val) { m_val = val; }
};
int main()
{ 
	Box<std::string> box1 = std::string("Love you");
	std::cout << box1.GetVal();
}
```
其实像`std::array`、`std::vector`都是类模板。
