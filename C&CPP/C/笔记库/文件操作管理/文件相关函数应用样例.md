## fputs()和fprintf()
这俩函数用来输出。
```C
#include <stdio.h>
int main() {
	FILE* p = fopen("test.txt","w");
	if (p == NULL) {
		perror("发生了未知错误");
		return 1;
	}
	fprintf(p,"fpirntf:%d\n", 2);
	fputs("fputs", p);
	fclose(p);
	p = NULL;
}
```
## fscanf()和feof()和ferror()
第一个用来格式化读取。第二个和第三个用来判断不读取的原因，请不要将他们作为读取循环的判断条件！他们是在读取结束之后用来寻找原因的！应该先判断是否空/到文件结尾，再判断是不是发生了IO错误。

```C
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int main() {
	FILE* p = fopen("input.txt","r");
	if (p == NULL) {
		perror("ERROR");
		return 1;
	}
	char str1[40]= { 0 };
	while (fscanf(p, "%39s", str1) == 1) {
		printf("%s\n", str1);
	}
	if (feof(p)) {
		printf("File is empty or end of file reached.\n");
	}
	else if (ferror(p)) {
		printf("Error reading file");
	}
	fclose(p);
}
```
## 补充：scanf("%s", str)的逻辑？
我们知道 `%s`用来输入一个字符串，当扫描输入流的时候，他会自动跳过所有的前导空白符，从第一个非空白符开始读取，直到文件结尾或者遇到下一个空白符。空白符将不会被读取进入。他会在下次读取的时候被当作前导空白而被忽略。该函数还会在末尾自动添加一个结束符，以使其成为一个合法的C字符串。
## fgets()
该函数用来读取整行，他的返回值是str指针。他会将换行符一并读入，所以我们用strcspn函数来进行替换。此函数会找到第二个参数中的任意部分在第一个参数中最早出现的位置（0开始）。所以我们可以借此来直接定位空格符位置并进行替换。
```C
	while (fgets(str1, sizeof(str1), p) != NULL) {
		str1[strcspn(str1, "\n")] = '\0';
		printf("Read string: %s\n", str1);
	}
```
## fgetc()&fputc()
```C
	int a = fgetc(p);
	printf("%c", a);
```
`fgetc()`的作用就是从输入流中读取一个字符。
```C
#include <stdio.h>
#include <string.h>
int main() {
	FILE* p = fopen("input.txt","a+");
	if (p == NULL) {
		perror("ERROR");
		return 1;
	}
	int ret = fputc('J', p);
	if (ret == EOF) {
		printf("ERROR");
	}
	else printf("%c", ret);
	fclose(p);
}
```
这个代码是`fputc()`的使用样例，请注意我们不要直接以%c输出返回值，因为很可能返回值是EOF！在写入的时候应该注意，使用的格式是非只读的！
## fwrite()
```C
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <string.h>
typedef struct {
	int id;
	char name[50];
	float score;
} Student;
int main() {
	FILE* p = fopen("test2.txt","wb");

	if (p == NULL) {
		perror("ERROR");
		return 1;
	}
	Student a[] = {
		{1, "Jim", 91},
		{2, "Noffer", 77}
	};
	int cnt = fwrite(a, sizeof(Student), 2, p);
	if(cnt == sizeof(a)/sizeof(a[0])) printf("success!");
	else printf("Failed!");
	fclose(p);
}
```
这是将结构体以二进制的方式写入到文件中，非常重要！
## fread()
```C
#include <stdio.h>
#include <string.h>
int main() {
	FILE* p = fopen("input.txt","rb");
	if (p == NULL) {
		perror("ERROR");
		return 1;
	}
	char str1[100] = { 0 };
	int nums = fread(str1, sizeof(char), 10, p);
	printf("%s", str1);
	fclose(p);
}
```
这就是普通的从二进制文本中读取！
```C
#include <stdio.h>
#include <string.h>
typedef struct {
	int id;
	char name[50];
	float score;
} Student;
int main() {
	FILE* p = fopen("test2.txt","rb");

	if (p == NULL) {
		perror("ERROR");
		return 1;
	}
	Student stu[100] = { 0 };
	size_t cnt = fread(stu, sizeof(Student), 100, p);
	printf("Already read %u structs!\n", cnt);
	for (size_t i = 0; i < cnt; i++) {
		printf("ID: %d, Name: %s, Score: %.2f\n",
			stu[i].id, stu[i].name, stu[i].score);
	}
	fclose(p);
}
```
这是将结构体从二进制文本中读入，也非常重要！！！！！
