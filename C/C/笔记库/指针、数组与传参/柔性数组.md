在结构体中，如果最后一个成员是一个位置大小的数组，那么就成为柔性数组成员。
## 原始样例程序
```C
#include <stdio.h>
#include <stdlib.h>
typedef struct Tp {
	int a;
	char b;
	int c[];
}Tp;
int main() {
	Tp* p = (Tp*)malloc(sizeof(Tp) + 60);
	if (!p) {
		perror("发生致命错误");
		return 1;
	}
	int i;
	for (i = 0; i < 15; i++) {
		p->c[i] = i;
	}
	for (i = 0; i < 15; i++) {
		printf("%d ", p->c[i]);
	}
	Tp* ptr = (Tp*)realloc(p, sizeof(Tp) + 80);
	if (!ptr) {
		perror("发生致命错误");
		free(p);
		p = NULL;
		return 1;
	}
	p = ptr;
	ptr = NULL;
	for (i = 0; i < 20; i++) {
		p->c[i] = i;
	}
	for (i = 0; i < 20; i++) {
		printf("%d ", p->c[i]);
	}
	free(p);
	p = NULL;
	return 0;
}
```
这是个样例程序。但是请注意，这个程序其实写的不太好。因为`sizeof(Tp)`没有算上柔性数组的大小。如果柔性数组发生了对齐的话，实际可用大小是要小于我们预估的空间的，所以这样是非常危险的。
## 修改
我们可以通过以下公式：
$$size\_t\ size = offsetof\ (Struct, flexible\_array) + needed\_bytes;$$
来计算需要的大小。具体意思就是我们首先要知道柔性数组的偏移量，再加上所需要的大小。
```C
#include <stdio.h>
#include <stddef.h>
struct MyStruct
{
	int a;
	char b;
	int c;
	int d[];
};
int main() {
	printf("%d",offsetof(struct MyStruct,d));
	printf("%u",sizeof(struct MyStruct));
}
```
我们会发现柔性数组偏移量和结构体大小相同。
### 修改后的版本
```C
#include <stdio.h>
#include <stdlib.h>
#include <stddef.h>
typedef struct Tp {
	int a;
	char b;
	int c[];
}Tp;
int main() {
	const int COUNT_1 = 15;
	Tp* p = (Tp*)malloc(offsetof(Tp, c) + sizeof(int) * COUNT_1);
	if (!p) {
		perror("发生致命错误");
		return 1;
	}
	int i;
	for (i = 0; i < COUNT_1; i++) {
		p->c[i] = i;
	}
	for (i = 0; i < COUNT_1; i++) {
		printf("%d ", p->c[i]);
	}
	const int COUNT_2 = 20;
	Tp* ptr = (Tp*)realloc(p, offsetof(Tp, c) + sizeof(int) * COUNT_2);
	if (!ptr) {
		perror("发生致命错误");
		free(p);
		p = NULL;
		return 1;
	}
	p = ptr;
	ptr = NULL;
	for (i = 0; i < COUNT_2; i++) {
		p->c[i] = i;
	}
	for (i = 0; i < COUNT_2; i++) {
		printf("%d ", p->c[i]);
	}
	free(p);
	p = NULL;
	return 0;
}
```
