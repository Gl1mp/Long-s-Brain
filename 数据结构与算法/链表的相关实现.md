#include <stdio.h>
#include <stdlib.h>
typedef struct ListNode {
	int val;
	struct ListNode* next;
}ListNode;
ListNode* CreateNode(int val) {
	ListNode* node;
	node = (ListNode*)malloc(sizeof(ListNode));
	node->next = NULL;
	node->val = val;
	return node;
}
void Print(ListNode* head) {
	ListNode* p = head;
	while (p != NULL) {
		printf("%d ", p->val);
		p = p->next;
	}
	printf("\n");
}
void InsertNode(ListNode* pre_node, ListNode* cur_node) {
	cur_node->next = pre_node->next;
	pre_node->next = cur_node;
}
void DelNextNode(ListNode* cur_node) {
	if (cur_node == NULL || cur_node->next == NULL) return;

	ListNode* tmp = cur_node->next;      // 保存要删除的节点
	cur_node->next = tmp->next;          // 绕过要删除的节点
	free(tmp);                           // 释放内存！
}
int FindNode(ListNode* head, int val) {
	int index = 0;
	ListNode* p = head;
	while (p != NULL) {
		if(p->val == val) return index;
		p = p->next;
		index++;
	}
	return -1;
}
int Traverse(ListNode* head) {
	ListNode* p = head;
	int cnt = 0;
	while (p != NULL) {
		cnt++;
		p = p->next;
	}
	return cnt;
}
int main() {
	ListNode* n0 = CreateNode(1);
	ListNode* n1 = CreateNode(3);
	ListNode* n2 = CreateNode(2);
	ListNode* n3 = CreateNode(5);
	ListNode* n4 = CreateNode(4);
	n0->next = n1;
	n1->next = n2;
	n2->next = n3;
	n3->next = n4;
	Print(n0);
	ListNode* n_insert = CreateNode(99);
	InsertNode(n2, n_insert);
	Print(n0);
	DelNextNode(n2);
	Print(n0);
	int cnt = Traverse(n0);
	printf("NUM:%d\n", cnt);
	int index = FindNode(n0, 2);
	printf("INDEX:%d\n",index);
	ListNode* p = n0;
	while (p != NULL) {
		ListNode* tmp = p;
		p = p->next;
		free(tmp);
	}
}
