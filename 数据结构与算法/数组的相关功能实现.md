#include <stdio.h>
#include <time.h>
#include <string.h>
#include <stdlib.h>
int randomAccess(int* p, int size) {
	//以O（1）时间访问数组中的元素
	srand((unsigned int)time(NULL));
	int randnum = rand()%size;
	return *(p + randnum);
}
int* insertNumber(int* p, int size, int num, int index) {
	for (int i = size - 1; i > index; i--) {
		p[i] = p[i-1];
	}
	p[index] = num;
	return p;
}
int delNumber(int* p, int size, int index) {
	for (int i = index; i < size-1; i++) {
		p[i] = p[i + 1];
	}
	return size-1;
}
int traverse(int* p, int size) {
	int cnt = 0;
	for (int i = 0; i < size; i++) {
		cnt++;
	}
	return cnt;
}
int findNumber(int* p, int size, int num) {
	int index = -1;
	for (int i = 0; i < size; i++) {
		if(num == p[i]) index = i;
	}
	return index;
}
int* enlargeArray(int* p, int size, int enlarge_num) {
	int* np = (int*)calloc(enlarge_num+size,sizeof(int));
	if(np == NULL) return ;
	for (int i = 0; i < size; i++) {
		np[i] = p[i];
	}
	return np;
}

	insertNumber(a, 10, 999, 3);
	len = delNumber(a, 10, 3);
	int i;
	for (i = 0; i < len; i++) {
		printf("%d ", a[i]);
	}
	printf("\n");
	int cnt = traverse(a, len);
	printf("Num:%d\n", cnt);
	int index_find = findNumber(a, len, 999);
	printf("index_find:%d\n", index_find);
	int* array_enlarge = enlargeArray(a, len, 30);
	len = len + 30;
	for (i = 0; i < len; i++) {
		printf("%d ", array_enlarge[i]);
	}
	free(array_enlarge);
	array_enlarge = NULL;
}
